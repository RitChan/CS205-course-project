20 template<typename T>
21 Matrix<T>::Matrix(Matrix::MatrixInitList init_list): d1size(init_list.size()), d2size(0), _valid(true) {
22  int i = 0;
23  int j = 0;
24  for (const std::initializer_list<T> &row: init_list) {
25  if (row.size() <= 0 || (d2size > 0 && row.size() != d2size)) {
26  _valid = false;
27  d1size = 0;
28  d2size = 0;
29  entry = std::vector<std::vector<T>>();
30  break;
31  }
32  if (d2size != row.size()) {
33  d2size = row.size();
34  entry = std::vector<std::vector<T>>(d1size, std::vector<T>(d2size, T{}));
35  }
36  d2size = row.size();
37  j = 0;
38  for (T elem: row)
39  entry[i][j++] = elem;
40  i++;
41  }
42 }

113 Matrix<T> Matrix<T>::operator*(const Matrix<T> &other) const {
114  if (d2size != other.d1size)
115  return Matrix<T>();
116  int align_len = d2size;
117  Matrix<T> ret(d1size, other.d2size);
118  for (int i = 0; i < ret.d1size; i++) {
119  for (int j = 0; j < ret.d2size; j++) {
120  ret[i][j] = 0;
121  for (int p = 0; p < align_len; p++) {
122  ret[i][j] += entry[i][p] * other[p][j];
